Th√©oreme central limite

1,2. Loi uniforme

a=2
b=6
mu=.5*(a+b)
n=100
k=1000
Z=0
for (j in 1:k) {
  X=runif(n,a,b)
  Z[j]=sum(X)/n
}
hist(Z,probability=TRUE,breaks=50,main="Distribution de la moyenne empirique",xlab="Valeur",ylab="Densit√©")
variance=(b-a)*(b-a)/12
NX = seq(min(Z),max(Z),length.out=500)  
NY = dnorm(NX, mean = mu, sd=sqrt(variance/n)) 
lines(x=NX,y=NY,col="red")
